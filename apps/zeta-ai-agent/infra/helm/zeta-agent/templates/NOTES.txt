1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "zeta-agent.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "zeta-agent.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "zeta-agent.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "{{ include "zeta-agent.selectorLabels" . | replace "\n" "," }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Check the deployment status:
  kubectl get deployment {{ include "zeta-agent.fullname" . }} -n {{ .Release.Namespace }}
  kubectl get pods -l "{{ include "zeta-agent.selectorLabels" . | replace "\n" "," }}" -n {{ .Release.Namespace }}

3. View application logs:
  kubectl logs -f deployment/{{ include "zeta-agent.fullname" . }} -n {{ .Release.Namespace }}

4. Check application health:
  kubectl exec -it deployment/{{ include "zeta-agent.fullname" . }} -n {{ .Release.Namespace }} -- curl http://localhost:{{ .Values.service.targetPort }}/health

{{- if .Values.metrics.enabled }}
5. Access metrics:
  kubectl port-forward service/{{ include "zeta-agent.fullname" . }} {{ .Values.metrics.port }}:{{ .Values.metrics.port }} -n {{ .Release.Namespace }}
  # Then visit http://localhost:{{ .Values.metrics.port }}/metrics
{{- end }}

{{- if .Values.persistence.enabled }}
6. Check persistent storage:
  kubectl get pvc {{ include "zeta-agent.pvcName" . }} -n {{ .Release.Namespace }}
{{- end }}

{{- if .Values.autoscaling.enabled }}
7. Monitor auto-scaling:
  kubectl get hpa {{ include "zeta-agent.fullname" . }} -n {{ .Release.Namespace }}
  kubectl describe hpa {{ include "zeta-agent.fullname" . }} -n {{ .Release.Namespace }}
{{- end }}

8. Run tests:
  helm test {{ .Release.Name }} -n {{ .Release.Namespace }}

9. Troubleshooting:
  # Check events
  kubectl get events --sort-by=.metadata.creationTimestamp -n {{ .Release.Namespace }}
  
  # Describe the deployment
  kubectl describe deployment {{ include "zeta-agent.fullname" . }} -n {{ .Release.Namespace }}
  
  # Check service endpoints
  kubectl get endpoints {{ include "zeta-agent.fullname" . }} -n {{ .Release.Namespace }}

{{- if .Values.serviceMonitor.enabled }}
10. Prometheus monitoring:
  # The ServiceMonitor {{ include "zeta-agent.fullname" . }} has been created
  # Verify it's being picked up by Prometheus:
  kubectl get servicemonitor {{ include "zeta-agent.fullname" . }} -n {{ .Values.serviceMonitor.namespace | default .Release.Namespace }}
{{- end }}
